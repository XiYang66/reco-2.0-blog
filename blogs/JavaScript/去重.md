## reduce去重

- 思路：一个目标数组对象，一个空对象，通过判断空对象中是否有数组中的id，有的话不做任何操作，没有则向对象中添加并push数组对象

```js
  let person = [
      { id: 0, name: '小明' },
      { id: 1, name: '小张' },
      { id: 2, name: '小李' },
      { id: 3, name: '小孙' },
      { id: 1, name: '小周' },
      { id: 2, name: '小陈' },
    ]
  let obj = {};
  let pen=person.reduce((cur,next)=>{
    obj[next.id]?'':obj[next.id]=true && cur.push(next)
    return cur
  },[])
```
